###Gruntfile.js

    module.exports = function (grunt) {
        // LiveReload的默认端口号，你也可以改成你想要的端口号
        var lrPort = 35729;
        // 使用connect-livereload模块，生成一个与LiveReload脚本
        var lrSnippet = require('connect-livereload')({ port: lrPort });
        // 使用 middleware(中间件)，就必须关闭 LiveReload 的浏览器插件
        var lrMiddleware = function (connect, options) {
            return [
                // 把脚本，注入到静态文件中
                lrSnippet,
                // 静态文件服务器的路径
                connect.static(options.base),
                // 启用目录浏览(相当于IIS中的目录浏览)
                connect.directory(options.base)
            ];
        };
        //使用 matchdep，可以自动载入所有任务。
        require("matchdep").filterDev("grunt-*").forEach(grunt.loadNpmTasks);
        // 构建任务配置
        grunt.initConfig({
            //读取package.json的内容，形成个json数据
            pkg: grunt.file.readJSON('package.json'),
            htmlhint: {
                build: {
                    options: {
                        'tag-pair': true,
                        'tagname-lowercase': true,
                        'attr-lowercase': true,
                        'attr-value-double-quotes': true,
                        'doctype-first': true,
                        'spec-char-escape': true,
                        'id-unique': true,
                        'head-script-disabled': true,
                        'style-disabled': true
                    },
                    src: ['assets/im-tiny.html']
                }
            },

            //grunt-cssc任务可以整合CSS文件中定义的样式规则，最大限度削减所生成的CSS文件中的重复内容。
            cssc: {
                build: {
                    options: {
                        consolidateViaDeclarations: true,
                        consolidateViaSelectors: true,
                        consolidateMediaQueries: true
                    },
                    files: {
                        'build/css/base.min.css': 'build/css/base.min.css'
                    }
                }
            },
            cssmin: {
                build: {
                    options: {
                        banner: '/*! <%= pkg.name %> - v<%= pkg.version %> - <%= grunt.template.today("yyyy-mm-dd") %> */\n'
                    },
    //                src: 'assets/css/base.css',
    //                dest: 'build/css/base.min.css'
                    files: [
                        {src: ['assets/css/base.css', 'assets/css/im.tiny.css'], dest: 'build/css/i.min.css'}
                    ]
                }
            },
            imagemin: {
                build: {
                    options: {
                        optimizationLevel: 3
                    },
                    files: {
                        src: 'assets/image/a.png',
                        dest: 'build/image/a.png'
                    }
                }
            },
            htmlmin: {
                build: {
                    options: {
                        removeComments: true,
                        collapseWhitespace: true
                    },
                    files: {
                        'build/index.html': 'assets/im-tiny.html'
                    }
                }
            },
            // 通过connect任务，创建一个静态服务器
            connect: {
                options: {
                    // 服务器端口号
                    port: 8000,
                    // 服务器地址(可以使用主机名localhost，也能使用IP)
                    hostname: 'localhost',
                    // 物理路径(默认为. 即根目录)
                    base: '.'
                },
                livereload: {
                    options: {
                        // 通过LiveReload脚本，让页面重新加载。
                        middleware: lrMiddleware
                    }
                }
            },
            concat: {
                build: {
                    options: {
                        separator: '\n;'
                    },
                    src: [ 'assets/js/*.js' ],
                    dest: 'build/js/i-min.js'
                }
            },
            uglify: {
                options: {
                    banner: '/*! <%= pkg.title || pkg.name %> - v<%= pkg.version %> - ' +
                        '<%= grunt.template.today("yyyy-mm-dd") %>\n' +
                        '<%= pkg.homepage ? "* " + pkg.homepage + "\\n" : "" %>' +
                        '* Copyright (c) <%= grunt.template.today("yyyy") %> <%= pkg.author.name %>;' +
                        ' Licensed <%= _.pluck(pkg.licenses, "type").join(", ") %> */\n',
                    mangle: false
                },
                build: {
                    files: {
                        'build/js/imLoadFile.min.js': ['assets/js/imLoadFile.js'],
                        'build/js/jquery.im.min.js': ['assets/js/jquery.im.js'],
                        'build/js/analyzeTemplet.js': ['assets/js/analyzeTemplet.js']
                    }
                },
                dest: {
                    files: {
                        'build/js/i.min.js': ['build/js/i.min.js']
                    }
                }

            },
            clean: {
                build: {
                    src: [ 'build/js', 'build/css' ]
                }
            },
            compress: {
                main: {
                    options: {
                        archive: 'archive.zip'
                    },
                    files: [
                        {src: 'build/css/*', dest: 'build/css', filter: 'isFile'},
                        {src: 'build/**', dest: 'build/'}
                    ]
                }
            },
            less: {
                development: {
                    options: {
                        paths: ["assets/css"]
                    },
                    files: {
                        "build/css/result.css": "assets/css/source.less"
                    }
                }
            },
            qunit: {
                files: ['test/*.html']
            },
            watch: {
                html: {
                    files: ['assets/*.html'],
                    tasks: ['htmlhint']
                },
                js: {
                    files: ['assets/js/*.js'],
                    tasks: ['uglify']
                },
                css: {
                    files: ['assets/css/*.css'],
                    tasks: ['buildcss']
                },
                concat: {
                    files: ['assets/js/*.js'],
                    tasks: ['concat']
                },
                client: {
                    options: {
                        livereload: lrPort
                    },
                    files: ['assets/im-tiny.html', 'assets/css/*', 'assets/js/*', 'assets/images/*']
                }
            }
        });

        // 默认执行的任务
        grunt.registerTask('default', []);
        // 自定义任务
        grunt.registerTask('live', ['connect', 'watch']);
        grunt.registerTask('buildcss', ['less', 'cssc', 'cssmin']);
    };
